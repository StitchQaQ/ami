func countMaxLen(s string) int {
	hashmap := make(map[rune]int)
	left, count := 0, 0

	for i, char := range s {
		if leftIdx, ok := hashmap[char]; ok && leftIdx >= left {
			left = leftIdx + 1
		}
		hashmap[char] = i

		//
		curLen := i - left + 1
		if curLen > count {
			count = curLen
		}
	}

	return count

}

func Fib(n int) <-chan int {

    ch := make(chan int)
    a, b := 0,1

    go func () {
        defer close(ch)

        for i := 0; i < n; i++ {
            ch <- a
            a, b = b, a+b
        }

    }()

    return ch
}
